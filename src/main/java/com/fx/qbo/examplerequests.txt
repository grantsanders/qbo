 URIBuilder builder = new URIBuilder()                          // builds query uri with sql query
      .setScheme("https")
      .setHost("sandbox-quickbooks.api.intuit.com")
      .setPath("/v3/company/" + realmId + "/query")
      .setParameter("query", "select * from customer where displayname = 'exampleName'");
      String query = builder.toString();

      HttpClient HTTPClient = HttpClient.newHttpClient();       // example HttpClient for query request
      HttpRequest request = HttpRequest.newBuilder()
      .uri(URI.create(query))
      .header("Authorization", "Bearer " + accessToken)
      .header("content_type", "application/json")
      .header("Accept", "application/json")
      .GET()
      .build();
      HTTPClient.sendAsync(request, BodyHandlers.ofString())
      .thenApply(HttpResponse::body)
      .thenAccept(System.out::println)
      .join();

      HttpClient post = HttpClient.newHttpClient();             // example HttpClient for post request
      HttpRequest postRequest = HttpRequest.newBuilder()
          .uri(URI.create(baseURL + "/v3/company/" + realmId + "/customer/"))
          // .uri(URI.create("https://sandbox-quickbooks.api.intuit.com/v3/company/4620816365236753060/customer"))
          .header("Authorization", "Bearer " + accessToken)
          .header("content-type", "application/json")
          .header("accept", "application/json")
          .POST(BodyPublishers.ofString("{\"DisplayName\":\"api test\"}"))
          .build();
      post.sendAsync(postRequest, BodyHandlers.ofString())
          .thenApply(HttpResponse::body)
          .thenAccept(System.out::println)
          .join();
